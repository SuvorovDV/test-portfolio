  import { useCallback, useState } from 'react';
  import { useReducedMotion } from './useReducedMotion';

  const STORAGE_KEY = 'portfolio:boot-shown';

  function readFlag(): boolean {
    if (typeof window === 'undefined') return true;
    try {
      return window.localStorage.getItem(STORAGE_KEY) === '1';
    } catch {
      // localStorage может быть заблокирован (privacy mode) — считаем что показан,
      // чтобы не зацикливать бут на каждой перезагрузке.
      return true;
    }
  }

  function writeFlag(): void {
    try {
      window.localStorage.setItem(STORAGE_KEY, '1');
    } catch {
      /* no-op */
    }
  }

  export function useBootShown(): { shown: boolean; markShown: () => void } {
    const reduced = useReducedMotion();
    const [shown, setShown] = useState(() => reduced || readFlag());

    const markShown = useCallback(() => {
      writeFlag();
      setShown(true);
    }, []);

    return { shown, markShown };
  }
  ```

  ВАЖНО:
  - Lazy init через callback в `useState` — readFlag вызывается один раз при mount
  - `reduced || readFlag()` — при reduced сразу shown=true (не играем бут вообще)
  - При reduced НЕ пишем в localStorage — вдруг пользователь потом выключит reduced-motion и захочет увидеть бут
  - try/catch вокруг localStorage — privacy mode, ITP и т.д.

  import { useEffect, useRef } from 'react';
  import { Cursor } from './Cursor';
  import styles from './BootSequence.module.css';

  const LINES: readonly string[] = [
    'Detecting hardware...',
    '  [ OK ] Display',
    '  [ OK ] Phosphor CRT @ 33FF33',
    '  [ OK ] Scanlines emulation',
    '',
    'Loading modules...',
    '  about.mod       [ OK ]',
    '  projects.mod    [ OK ]',
    '  contact.mod     [ OK ]',
  ];

  const STEP_MS = 180;
  const FINAL_PAUSE_MS = 500;

  interface BootSequenceProps {
    onFinish: () => void;
  }

  export function BootSequence({ onFinish }: BootSequenceProps) {
    const onFinishRef = useRef(onFinish);
    onFinishRef.current = onFinish;

    useEffect(() => {
      const totalMs = LINES.length * STEP_MS + FINAL_PAUSE_MS;
      const autoFinish = window.setTimeout(() => onFinishRef.current(), totalMs);

      const handleKey = (e: KeyboardEvent) => {
        // Любая клавиша завершает бут
        e.preventDefault();
        onFinishRef.current();
      };
      const handleClick = () => onFinishRef.current();

      window.addEventListener('keydown', handleKey);
      window.addEventListener('click', handleClick);

      return () => {
        window.clearTimeout(autoFinish);
        window.removeEventListener('keydown', handleKey);
        window.removeEventListener('click', handleClick);
      };
    }, []);

    return (
      <div className={styles.boot} role="status" aria-live="polite" aria-atomic="false">
        <button
          type="button"
          className={styles.skip}
          onClick={(e) => {
            // Click handler на window уже поймает это. Но explicit кнопка — для tab-доступа.
            e.stopPropagation();
            onFinishRef.current();
          }}
          aria-label="Skip boot sequence"
        >
          [ skip ]
        </button>

        <header className={styles.header}>
          <span className={styles.line} style={{ animationDelay: '0ms' }}>
            TEST-PORTFOLIO BIOS v1.0
          </span>
          <span className={styles.line} style={{ animationDelay: '100ms' }}>
            Copyright (c) 2026 Suvorov Denis
          </span>
        </header>

        {LINES.map((text, i) => (
          <span
            key={i}
            className={`${styles.line} ${text.includes('[ OK ]') ? styles.ok : ''}`}
            style={{ animationDelay: `${(i + 2) * STEP_MS}ms` }}
          >
            {text || '\u00A0'}
          </span>
        ))}

        <span
          className={`${styles.line} ${styles.final}`}
          style={{ animationDelay: `${(LINES.length + 2) * STEP_MS}ms` }}
        >
          Boot successful. Press any key to continue
          <Cursor />
        </span>
      </div>
    );
  }
  ```

  ВАЖНО:
  - `onFinishRef` паттерн — listener'ы читают свежий callback без реинициализации effect'а
  - `animationDelay` inline через index → каждая строка fade'ится на своём timing'е без JS-state per-line
  - Empty string `''` заменён на `'\u00A0'` (nbsp) чтобы высота строки сохранялась
  - `role="status" aria-live="polite" aria-atomic="false"` — SR читает по строкам, не всё целиком
  - Button `[ skip ]` с aria-label, stopPropagation чтобы не двойной-тригер вместе с window.click
  - window.click/keydown на уровне window — любой клик/клавиша завершают бут
  - Пустая зависимость `[]` в useEffect — listener'ы вешаются один раз, onFinish через ref
