  Изменить `src/App.tsx`:

  ```tsx
  import { AboutSection } from './sections/AboutSection';
  import { ProjectsSection } from './sections/ProjectsSection';
  import { ContactSection } from './sections/ContactSection';
  import { CRTShell } from './components/CRTShell';
  import { TerminalPrompt } from './components/TerminalPrompt';
  import { BootSequence } from './components/BootSequence';
  import { useBootShown } from './hooks/useBootShown';
  import { useHashNavigation } from './hooks/useHashNavigation';

  export function App() {
    const { shown, markShown } = useBootShown();
    useHashNavigation();

    if (!shown) {
      return <BootSequence onFinish={markShown} />;
    }

    return (
      <CRTShell>
        <TerminalPrompt />
        <main>
          <AboutSection />
          <ProjectsSection />
          <ContactSection />
        </main>
      </CRTShell>
    );
  }
  ```

  ВАЖНО:
  - `useHashNavigation()` остаётся всегда вызванным (правила hooks), даже во время boot. Хук не имеет side-effects когда нет элементов с id — безопасно.
  - Условный рендер после вызова всех хуков — иначе нарушение rules-of-hooks
  - На первом рендере `shown` уже инициализирован из хука (lazy init), не будет вспышки контента перед бутом
