  ```tsx
  import { useEffect, useState } from 'react';
  import { useReducedMotion } from '../hooks/useReducedMotion';
  import styles from './TypedText.module.css';

  interface TypedTextProps {
    text: string;
    /** Delay between characters in ms. Default 40. */
    speed?: number;
  }

  export function TypedText({ text, speed = 40 }: TypedTextProps) {
    const reduced = useReducedMotion();
    const [displayed, setDisplayed] = useState(reduced ? text : '');

    useEffect(() => {
      if (reduced) {
        setDisplayed(text);
        return;
      }
      let i = 0;
      setDisplayed('');
      const timer = setInterval(() => {
        i += 1;
        setDisplayed(text.slice(0, i));
        if (i >= text.length) clearInterval(timer);
      }, speed);
      return () => clearInterval(timer);
    }, [text, speed, reduced]);

    return (
      <>
        {/* Visual — typewritten, aria-hidden to avoid SR re-reading partial strings */}
        <span className={styles.visual} aria-hidden="true">
          {displayed}
        </span>
        {/* SR-only — full text read once */}
        <span className={styles.sr}>{text}</span>
      </>
    );
  }
  ```

  ВАЖНО:
  - Visual aria-hidden + sr-only полный текст — screen reader читает один раз полную фразу, не partial ('S', 'Su', 'Suv', ...)
  - cleanup clearInterval в return эффекта — предотвращает утечку в StrictMode (double-mount)
  - При reduced=true сразу устанавливаем полный text, без interval'а
  - НЕ рендерим Cursor здесь — caller сам решит нужен ли
  </action>

  **CopyButton.tsx:**
  ```tsx
  import { useState } from 'react';
  import styles from './CopyButton.module.css';

  interface CopyButtonProps {
    value: string;
    label?: string;
    copiedLabel?: string;
  }

  export function CopyButton({
    value,
    label = '[copy]',
    copiedLabel = '[copied]',
  }: CopyButtonProps) {
    const [copied, setCopied] = useState(false);

    const handleClick = async () => {
      try {
        await navigator.clipboard.writeText(value);
        setCopied(true);
        setTimeout(() => setCopied(false), 2000);
      } catch {
        // clipboard API может отказать (не secure context, нет прав).
        // В этом случае — просто игнорируем, пользователь увидит что label не изменился.
      }
    };

    return (
      <button
        type="button"
        onClick={handleClick}
        className={`${styles.button} ${copied ? styles.copied : ''}`}
        aria-live="polite"
      >
        {copied ? copiedLabel : label}
      </button>
    );
  }
  ```

  ВАЖНО:
  - `aria-live="polite"` — SR озвучит смену label на [copied]
  - `type="button"` — важно, иначе внутри формы submit'ит
  - setTimeout 2000мс reset
  - try/catch на clipboard (может упасть на не-secure origin)
  </action>
